集合栈计算机

Posted on 2019-12-05

题目：https://vjudge.net/problem/UVA-12096

【分析】：(算法竞赛入门经典第二版P116)
本题的集合实际是集合的集合，每个集合都有唯一的ID,因此任意一个集合通过包含多个ID可以让多个其他集合成为它的内部元素。
集合类型：set<int> ID类型：int

#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <map>
#include <stack>
#include <fstream>
using namespace std;

typedef set<int> Set; 
vector<Set> v; //存储各个集合，并为其分配ID
map<Set,int> my_map; //集合与对应的ID做映射，根据ID取集合
stack<int> st; //st中为集合的ID

int get_ID(Set x)
{
    if(my_map.count(x))  return my_map[x];
    v.push_back(x);
    my_map[x] = v.size()-1;
    return v.size()-1;
}

void my_set_add(Set& s,Set& s1,Set& s2)
{
    s = s2;
    s.insert(my_map[s1]);
    int id = get_ID(s);
    st.push(id);
}
void my_set_intersection(Set& s,Set& s1,Set& s2)
{
    auto s1_begin = s1.begin(), s1_end = s1.end();
    auto s2_begin = s2.begin(), s2_end = s2.end();
    if(s1.empty() || s2.empty()) {s = Set(); int id = get_ID(s); st.push(id); return;}
    vector<int> temp(int(s1.size())+int(s2.size()));
    set_intersection(s1_begin,s1_end,s2_begin,s2_end,temp.begin());
    s = set<int>(temp.begin(),temp.end());
    int id = get_ID(s);
    st.push(id);
}
void my_set_union(Set& s,Set& s1,Set& s2)
{
    auto s1_begin = s1.begin(), s1_end = s1.end();
    auto s2_begin = s2.begin(), s2_end = s2.end();
    vector<int> temp(int(s1.size())+int(s2.size()));
    set_union(s1_begin,s1_end,s2_begin,s2_end,temp.begin());
    s = set<int>(temp.begin(),temp.end());
    int id = get_ID(s);
    st.push(id);
}

int main()
{
    //将输出重定向到文件里便于检查
    // fstream fs;
    // fs.open("test.txt",ios::out);

    int n;
    cin >> n;
    for(int i = 0; i < n; i++)
    {
        int num;
        cin >> num;
        for(int j = 0; j < num; j++)
        {
            string cmd; cin >> cmd;
            if(cmd[0] == 'P') st.push(get_ID(Set()));
            else if(cmd[0] == 'D') st.push(st.top());
            else
            {
                Set s1 = v[st.top()]; st.pop();
                Set s2 = v[st.top()]; st.pop();
                Set s;
                if(cmd[0] == 'U') my_set_union(s,s1,s2);
                if(cmd[0] == 'I') my_set_intersection(s,s1,s2);
                if(cmd[0] == 'A') my_set_add(s,s1,s2);
            }
            //fs << v[st.top()].size() << endl;
            cout << v[st.top()].size() << endl;
        }
        //fs << "***" << endl;
        cout << "***" << endl;
    }
    system("pause");
    //fs.close();
    return 0;
}

值得一提的是学习了set中求并集和交集的两个函数，set_union与set_intersection，函数定义见http://www.cplusplus.com/reference/algorithm/set_union/，http://www.cplusplus.com/reference/algorithm/set_intersection/。
函数的前4个参数为两个set的起始地址、结束地址，最后一个参数为保存结果的地址。
踩过的坑：保存结果的地址必须已分配。例如将结果保存在vector中，vector地址需要事先分配。

收获：
1.嵌套的思想(集合的集合)
2.set中并集交集函数的学习
3.fstream的使用，通过将输出写到文件里，便于oj上题目的调试
